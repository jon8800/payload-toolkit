```typescript
/**
 * Generate a static JSON file of Google Font family names from google-font-metadata.
 * Run: npx tsx scripts/generate-font-list.ts
 */
import { writeFileSync } from 'fs'
import { resolve } from 'path'

async function main() {
  // google-font-metadata exports functions to get font data
  // The APIv2 data includes all Google Fonts with metadata
  const { APIv2 } = await import('google-font-metadata')
  const data = await APIv2()

  // Extract font family names sorted alphabetically
  const families = Object.keys(data).sort((a, b) => a.localeCompare(b))

  const outputPath = resolve(__dirname, '../src/data/google-fonts.json')
  writeFileSync(outputPath, JSON.stringify(families, null, 0))

  console.log(`Generated ${families.length} font families to ${outputPath}`)
}

main().catch(console.error)
```

**Lazy font preview loading (D-31):**
Create a `useFontPreview` hook or inline effect that:
1. Tracks which fonts are currently visible in the dropdown viewport
2. Uses an IntersectionObserver on option elements
3. When a font option becomes visible, injects a `<link>` element into `<head>`:
   `<link href="https://fonts.googleapis.com/css2?family=FONT_NAME:wght@400&display=swap" rel="stylesheet">`
4. Each option renders with `style={{ fontFamily: '"FontName", sans-serif' }}` — before the CSS loads, it falls back to sans-serif; after load, it shows the actual font
5. Keep a Set of already-loaded fonts to avoid duplicate link injection
6. Clean up: do NOT remove links on unmount (fonts stay cached for the session)

**Implementation of font loading:**
```typescript
const loadedFontsRef = useRef<Set<string>>(new Set())

const loadFont = useCallback((fontName: string) => {
  if (loadedFontsRef.current.has(fontName)) return
  loadedFontsRef.current.add(fontName)
  const link = document.createElement('link')
  link.href = `https://fonts.googleapis.com/css2?family=${encodeURIComponent(fontName)}:wght@400&display=swap`
  link.rel = 'stylesheet'
  document.head.appendChild(link)
}, [])
```

**IntersectionObserver for visible options:**
```typescript
const listRef = useRef<HTMLDivElement>(null)

useEffect(() => {
  const listEl = listRef.current
  if (!listEl) return

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const fontName = (entry.target as HTMLElement).dataset.font
          if (fontName) loadFont(fontName)
        }
      })
    },
    { root: listEl, threshold: 0 }
  )

  listEl.querySelectorAll('[data-font]').forEach(el => observer.observe(el))
  return () => observer.disconnect()
}, [filteredFonts, loadFont])  // Re-observe when filtered list changes
```
