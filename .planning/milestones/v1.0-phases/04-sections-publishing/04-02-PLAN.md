export const atomicBlockSlugs = [
  'heading', 'paragraph', 'list', 'blockquote',
  'image', 'video', 'icon',

Every block has:

- `children`: nested blocks array (optional)

Grid-specific: `settings.columns` (number), `settings.gap` (none/xs/sm/md/lg/xl)
