2. Create `src/collections/Posts.ts`:
   - slug: 'posts'
   - access: create/delete/update = authenticated, read = authenticatedOrPublished
   - admin: defaultColumns: ['title', 'slug', 'updatedAt'], useAsTitle: 'title'
   - versions: { maxPerDoc: 50, drafts: { autosave: { interval: 300 }, schedulePublish: true } }
   - trash: true
   - enableQueryPresets: true
   - fields: title (text, required), slug (text, required, unique, sidebar), featuredImage (upload, relationTo: 'media'), excerpt (textarea), author (relationship to 'users', sidebar), categories (relationship to 'categories', hasMany, sidebar), tags (relationship to 'tags', hasMany, sidebar), publishedAt (date, sidebar), layout (blocks with blockReferences)

3. Create `src/collections/TemplateParts.ts`:
   - slug: 'template-parts'
   - access: create/delete/update = authenticated, read = authenticatedOrPublished
   - admin: defaultColumns: ['title', 'type', 'updatedAt'], useAsTitle: 'title'
   - versions: { maxPerDoc: 50, drafts: { autosave: { interval: 300 } } } (no schedulePublish needed for template parts)
   - trash: true
   - enableQueryPresets: true
   - fields:
     - title (text, required)
     - type (select, required, sidebar, options: Header/Footer/Custom)
     - displayCondition (group):
       - mode (select, defaultValue: 'entireSite', options: Entire Site, Specific Pages, Collection Type, Exclude Pages)
       - pages (relationship to 'pages', hasMany, conditionally shown when mode is specificPages or excludePages)
       - collectionType (select, options: All Pages/All Blog Posts, conditionally shown when mode is collectionType)
     - layout (blocks with blockReferences)

4. Create `src/globals/SiteSettings.ts`:
   - slug: 'site-settings'
   - access: read = anyone, update = authenticated
